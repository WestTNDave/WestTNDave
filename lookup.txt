var retStr = "";
var xhttp = new XMLHttpRequest();
xhttp.open("GET", "https://raw.githubusercontent.com/WestTNDave/WestTNDave/main/user_lookup.xml");
xhttp.send();
lookup(xhttp);

function lookup(xml) {
  var  xmlDoc = xml.responseXML;
  var  x = xmlDoc.getElementsByTagName("user");
  for (let i = 0; i <x.length; i++) {
    retStr +=
    "Twitch User " + 
    x[i].getElementsByTagName("twitch")[0].childNodes[0].nodeValue +
    " is " +
    x[i].getElementsByTagName("global")[0].childNodes[0].nodeValue +
    " on global poker. " + 
    x[i].getElementsByTagName("note")[0].childNodes[0].nodeValue +
    "\n";
  }
}

retStr;
